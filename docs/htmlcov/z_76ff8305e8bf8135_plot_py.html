21def smooth1D(data: NDArray, winSize: int = 5) -> NDArray: 

22 ''' Smooth an 1D array using moving average window. 

23 The implementation is from https://stackoverflow.com/questions/40443020 

24 ''' 

25 cripAssert(is1D(data), '`data` should be 1D array.') 

26 cripAssert(isInt(winSize) and winSize % 2 == 1, '`winSize` should be odd positive integer.') 

27 

28 out0 = np.convolve(data, np.ones(winSize, dtype=int), 'valid') / winSize 

29 r = np.arange(1, winSize - 1, 2) 

30 start = np.cumsum(data[:winSize - 1])[::2] / r 

31 stop = (np.cumsum(data[:-winSize:-1])[::2] / r)[::-1] 

32 

33 return np.concatenate((start, out0, stop))

42def window(img: TwoOrThreeD, 

43 win: Tuple[float, float], 

44 style: str = 'lr', 

45 normalize: Or[str, None] = None) -> TwoOrThreeD: 

46 ''' Window `img` using `win` (WW, WL) with style `wwwl` or (left, right) with style `lr`. 

47 Set `normalize` to `0255` to convert to 8-bit image, or `01` to [0, 1] float image. 

48 ''' 

49 cripAssert(len(win) == 2, '`win` should have length of 2.') 

50 cripAssert(style in ['wwwl', 'lr'], "`style` should be 'wwwl' or 'lr'") 

51 

52 if style == 'wwwl': 

53 ww, wl = win 

54 l = wl - ww / 2 

55 r = l + ww 

56 elif style == 'lr': 

57 l, r = win 

58 

59 res = asFloat(img.copy()) 

60 res[res > r] = r 

61 res[res < l] = l 

62 

63 if normalize == '0255': 

64 res = cv2.normalize(res, None, 0, 255, cv2.NORM_MINMAX).astype(np.uint8) 

65 elif normalize == '01': 

66 res = (res - l) / (r - l) 

67 

68 return res

71def windowFullRange(img: TwoOrThreeD, normalize='01') -> TwoOrThreeD: 

72 ''' Window `img` using its full dynamic range of pixel values. 

73 ''' 

74 return window(img, (np.max(img), np.min(img)), 'lr', normalize)
